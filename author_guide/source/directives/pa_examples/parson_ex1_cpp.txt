   Construct a block of code that correctly implements the accumulator pattern.
   -----
   template < class InputIt, class T >
   T sum(InputIt first, InputIt last, T value)
   {
       while (first != last) {
          value = value + *first;
          ++first;
       }
       return value;
   }
